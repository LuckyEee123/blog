创建一个Student类

package com.mss.HomeWork;

public class Student {
String name;
int id;
int age;
String sex;

public Student(String name, int id, int age, String sex) {
    this.name = name;
    this.id = id;
    this.age = age;
    this.sex = sex;
}

public Student() {
}

void show() {
    System.out.println("学生的名字是" + this.name + "学号id为" + this.id + "性别为" + this.sex + "，" + this.age + "岁");
}

public static void main(String[] args) {
    Student r1 = new Student("麦思帅", 52, 19, "男");
    r1.show();
}

}
